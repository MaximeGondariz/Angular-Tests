import { Component } from '@angular/core';

@Component({
  selector: 'app-pizza',
  templateUrl: './pizza.component.html',
  styleUrls: ['./pizza.component.scss']
})
export class PizzaComponent {

  NomPizza: Array<string> = ["Margarita","Chèvre Miel","Kebab","Ananas"]

  DetaillePizza0: any = {Titre : "Margarita",Ingredient : ["Sauce Tomate","Lardons","Emmentale"],Id : 0}
  DetaillePizza1: any = {Titre : "Chèvre Miel",Ingredient : ["Crème Fraiche","Lardons","Chèvre","Miel"],Id : 1}
  DetaillePizza2: any = {Titre : "Kebab",Ingredient : ["Sauce Tomate","Kebab","Emmentale"],Id : 2}
  DetaillePizza3: any = {Titre : "Ananas",Ingredient : ["Sauce Tomate","Lardons","Ananas","Emmentale"],Id : 3}

  DetaillePizza = this.DetaillePizza0

  onReceivedCD(IdDetaille: number){
    console.log(IdDetaille)
    switch(IdDetaille){
      case IdDetaille=0:{
        this.DetaillePizza = this.DetaillePizza0
        break;
      }
      case IdDetaille=1:{
        this.DetaillePizza = this.DetaillePizza1
        break;
      }
      case IdDetaille=2:{
        this.DetaillePizza = this.DetaillePizza2
        break;
      }
      case IdDetaille=3:{
        this.DetaillePizza = this.DetaillePizza3
        break;
      }
    }
  }
}
